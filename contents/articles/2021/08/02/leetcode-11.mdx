/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function(height) {
  let start = 0
  let end = height.length - 1
  let result = 0
  
  while (start < end) {
    const area = getArea(start, end, height);
    if (area > result) {
      result = area
    } else {
      // move up/down the index with the smaller height
      if (height[start] >= height[end]) {
        end--;
      } else {
        start++;
      }
    }
  }
  return result;
};

var getArea = function(start, end, array) {
  return (end - start) * Math.min(array[start], array[end])
}
