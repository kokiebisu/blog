```js
/**
 * @param {number} n
 * @return {string[]}
 */
var fizzBuzz = function (n) {
  var result = []

  var generate = number => {
    if (number % 5 === 0 && number % 3 === 0) {
      result.push("FizzBuzz")
    } else if (number % 5 === 0) {
      result.push("Buzz")
    } else if (number % 3 === 0) {
      result.push("Fizz")
    } else {
      result.push(number)
    }
  }
  for (let i = 0; i < n; i++) {
    generate(i)
  }
  return result
}
```
