Using Bitwise Operator

```js
/**
 * @param {number} n - a positive integer
 * @return {number}
 */
var hammingWeight = function (n) {
  let totalBits = 32
  let count = 0
  while (totalBits > 0) {
    if (n & 1) count++
    n = n >> 1
    totalBits--
  }
  return count
}
```
