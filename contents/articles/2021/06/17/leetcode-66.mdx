Brute Force Approach

```js
/**
 * @param {number[]} digits
 * @return {number[]}
 */
var plusOne = function (digits) {
  let count = 1
  for (let i = 0; i < digits.length; i++) {
    if (digits[digits.length - 1 - i] < 9) {
      digits[digits.length - 1 - i] = digits[digits.length - 1 - i] + count
      count = 0
      break
    } else {
      digits[digits.length - 1 - i] = 0
    }
  }
  if (count > 0) {
    digits.unshift(count)
  }
  return digits
}
```
