```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function (nums) {
  let i = 0
  let j = 1
  while (j <= nums.length - 1) {
    if (nums[i] !== nums[j]) {
      i++
      nums[i] = nums[j]
    } else {
      j++
    }
  }
  return i + 1
}
```
