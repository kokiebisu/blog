Key Issue

> How can we tell if we are in a infinite loop?

-> If we record a map with all the numbs that appeared and if it collides with whatever number it has in map

```js
/**
 * @param {number} n
 * @return {boolean}
 */
var isHappy = function (n) {
  const map = {}
  function inner(num) {
    for (let i = 0; i < nums.length; i++) {

    }
    if (sum === 1) {
      return true
    } else if (map[sum]) {
      return false
    } else {
      map[sum] = true
      inner(sum)
    }
  }
  inner(n)
}
```
