### Commonjs

```js
let something = null
if (isPerson) {
  something = require("./person-info")
} else {
  something = require("./creature-info")
}
```

### ESModules

```js
let something = null

if (isPerson) {
  await import("./person-info").then(data => {
    something = data
  })
} else {
  await import("./creature-info").then(data => {
    something = data
  })
}
```
