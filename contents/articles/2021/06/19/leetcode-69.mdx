Straightforward Approach

```js
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
  let max = 1
  while (x >= max * max) {
    max++
  }
  return max - 1
}
```

Optimized (Using Binary Search)

```js
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
  let left = 0
  let right = Math.floor(x / 2)
  let middle
  while (left <= right) {
    middle = Math.floor((left + right) / 2)
    const square = middle * middle
    if (square > x) right = middle - 1
    else if (square < x) left = middle + 1
    else return middle
  }
  return right
}
```
